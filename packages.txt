# Installation Guide for Fiber Length Calculator

# Step 1: Install Ollama
# For Linux/WSL:
curl -fsSL https://ollama.ai/install.sh | sh

# For Windows (download from https://ollama.ai/download)
# For macOS: brew install ollama

# Step 2: Install Python dependencies
pip install -r requirements.txt

# Step 3: Install NVIDIA CUDA Toolkit for GPU support (Linux)
# Ubuntu/Debian:
sudo apt update
sudo apt install nvidia-cuda-toolkit nvidia-driver-535

# For other systems, download CUDA from:
# https://developer.nvidia.com/cuda-downloads

# Step 4: Set environment variables (if needed)
export LD_LIBRARY_PATH=/usr/local/cuda/lib64:$LD_LIBRARY_PATH
export PATH=/usr/local/cuda/bin:$PATH

# Step 5: Verify GPU support
nvidia-smi

# Step 6: Start the application
streamlit run app.py

# Alternative manual setup:
# 1. Start Ollama service manually:
#    ollama serve
# 2. Pull the model:
#    ollama pull llama3.2-vision:11b
# 3. Run Streamlit:
#    streamlit run app.py

# Troubleshooting:
# - If GPU not detected, ensure CUDA drivers are properly installed
# - If Ollama fails to start, check if port 11434 is available
# - For Windows users, use WSL2 or install Ollama Desktop app
